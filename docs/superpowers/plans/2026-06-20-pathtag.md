> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 wd14-tagger-web 新增「路径打标」页签：用户输入文件夹路径，后端反推其中所有图片，把标签写为同名 `.txt`，支持 cl_tagger_v2。

**Architecture:** 后端新增 `pathtag` 异步任务队列 + SSE（复用 `backend/tasks/queue.py` 与 `backend/api/routes_batch.py` 模式），纯写 txt、不入图库；前端新增 `PathTagPage` 页签 + `usePathTag` composable，复用 `useTagger`。推理走现有 `get_tagger(model).tag_image()`。

**Tech Stack:** FastAPI + uvicorn + PIL + onnxruntime（后端）；Vue 3 + naive-ui（按需 import）+ vitest（前端）；pytest（后端）。

**Spec:** [docs/superpowers/specs/2026-06-20-pathtag-design.md](../specs/2026-06-20-pathtag-design.md)

## Global Constraints

- 不引入 `torch`/`cv2`/`pandas`；不 import mikazuki（仅借鉴 glob + 写 txt 思路）。
- 不写 `storage`/`meta`（不入图库）；文本纯逗号无权重，不走 `Classifier`。
- txt 写图片**同目录同名**：`a.png` → `a.txt`（`Path.with_suffix(".txt")`）。
- 推理签名统一：`tagger.tag_image(pil, gen_th, char_th, use_char) -> dict[str, float]`（v2 单阈值，`char_th` 仅接口一致保留）。
- 模型默认 `cl_tagger_v2`，阈值默认 v2→0.55/0.55、cl_tagger→0.35/0.6、wd14→0.35/0.9。
- 异步任务 + SSE：`asyncio.create_task` 起后台 `_run`，推理用 `asyncio.to_thread`，事件 `yield "data: " + json.dumps(evt) + "\n\n"`（对齐 `routes_batch.py`）。
- 后端测试：pytest + `fastapi.testclient.TestClient` + `FakeTagger`（`monkeypatch.setattr(deps, "get_tagger", ...)`）；图片用 `PIL.Image.new(...).save()` 造真图。
- 前端测试：vitest + jsdom + `@vue/test-utils`，`vi.stubGlobal('fetch', ...)`、mock `EventSource`。
- naive-ui **按需 import**（main.ts 未全局注册），组件在 `<script setup>` 显式 import。
- 所有提交直接进 `master` 分支（项目约定），中文 commit message。

```python
from __future__ import annotations
import logging
from glob import glob
from pathlib import Path

from PIL import Image

logger = logging.getLogger(__name__)


def _supported_extensions() -> set[str]:
    """PIL 能打开的图片扩展名集合（带点号、小写）。对齐 mikazuki interrogator.py 取法。"""
    return {e.lower() for e, f in Image.registered_extensions().items() if f in Image.OPEN}


def expand_images(path: str, recursive: bool = False) -> list[Path]:
    """把文件夹路径展开成按文件名排序的图片列表。
    recursive=False：仅直接子文件（path/*）；True：含子目录（path/**/*）。"""
    base = Path(path)
    pattern = f"{base}/**/*" if recursive else f"{base}/*"
    exts = _supported_extensions()
    paths = [
        Path(p) for p in glob(str(pattern), recursive=recursive)
        if Path(p).suffix.lower() in exts
    ]
    return sorted(paths, key=lambda p: p.name)
```

```python
def process_one(
    path: Path, tagger, gen_th: float, char_th: float,
    use_char: bool, on_conflict: str,
) -> tuple[str, dict]:
    """处理单张图：读图 → 反推 → 写同名 .txt。返回 (status, sse_event_partial)。
    status ∈ {"ok","skip","error"}；event 已含 type/current/(status|message)，
    done/total 由调用方（_run）在 append 前补全。单张失败不抛出，交由调用方累计。"""
    try:
        with Image.open(path) as pil:
            tags = tagger.tag_image(pil, gen_th, char_th, use_char)
        # tag_image 已做阈值过滤 + _→空格，这里仅按分数降序拼成纯逗号文本（无权重）。
        text = ", ".join(sorted(tags, key=tags.get, reverse=True))
        out = path.with_suffix(".txt")
        if on_conflict == "skip" and out.exists():
            return "skip", {"type": "progress", "current": path.name, "status": "skip"}
        out.write_text(text, encoding="utf-8")
        return "ok", {"type": "progress", "current": path.name, "status": "ok"}
    except Exception as e:
        logger.warning("pathtag %s failed: %r", path, e)
        return "error", {"type": "error", "current": path.name, "message": str(e)}
```

```python
MAX_JOBS = 16


@dataclass
class PathTagState:
    job_id: str
    total: int
    done: int = 0
    ok: int = 0
    skipped: int = 0
    failed: int = 0
    finished: bool = False
    events: list[dict] = field(default_factory=list)
    waiters: list[asyncio.Event] = field(default_factory=list)


class PathTagQueue:
    """异步批量路径打标队列。结构对齐 backend/tasks/queue.py 的 BatchQueue。"""

    def __init__(self):
        self._jobs: "OrderedDict[str, PathTagState]" = OrderedDict()

    def submit(self, path: str, model: str, gen_th: float, char_th: float,
               use_char: bool, recursive: bool, on_conflict: str) -> tuple[str, int]:
        images = expand_images(path, recursive)
        job_id = secrets.token_hex(8)
        state = PathTagState(job_id=job_id, total=len(images))
        self._jobs[job_id] = state
        while len(self._jobs) > MAX_JOBS:
            self._jobs.popitem(last=False)
        asyncio.create_task(self._run(state, images, model, gen_th, char_th,
                                      use_char, on_conflict))
        return job_id, len(images)

    async def _run(self, state, images, model, gen_th, char_th, use_char, on_conflict):
        tagger = deps.get_tagger(model)
        for p in images:
            status, partial = await asyncio.to_thread(
                process_one, p, tagger, gen_th, char_th, use_char, on_conflict)
            if status == "ok":
                state.ok += 1
            elif status == "skip":
                state.skipped += 1
            else:
                state.failed += 1
            evt = dict(partial)
            if evt.get("type") == "progress":
                evt["done"] = state.done + 1
                evt["total"] = state.total
            state.events.append(evt)
            state.done += 1
            for w in state.waiters:
                w.set()
            state.waiters.clear()
        state.finished = True
        state.events.append({"type": "done", "done": state.done, "total": state.total,
                             "ok": state.ok, "skipped": state.skipped, "errors": state.failed})
        for w in state.waiters:
            w.set()
        state.waiters.clear()

    def state(self, job_id):
        return self._jobs.get(job_id)


_pathtag_queue: PathTagQueue | None = None


def get_pathtag_queue() -> PathTagQueue:
    global _pathtag_queue
    if _pathtag_queue is None:
        _pathtag_queue = PathTagQueue()
    return _pathtag_queue
```

- [ ] **Step 3b: 实现路由** —— 新建 `backend/api/routes_pathtag.py`：

```python
import asyncio
import json
from pathlib import Path

from fastapi import APIRouter, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel

from backend.tasks.pathtag import get_pathtag_queue
from backend.tagger.models_spec import MODEL_SPECS

router = APIRouter(prefix="/api/pathtag", tags=["pathtag"])


class PathTagRequest(BaseModel):
    path: str
    model: str = "cl_tagger_v2"
    gen_th: float = 0.55
    char_th: float = 0.55
    use_char: bool = True
    recursive: bool = False
    on_conflict: str = "overwrite"


@router.post("/start")
async def start_pathtag(req: PathTagRequest):
    if req.model not in MODEL_SPECS:
        raise HTTPException(status_code=400, detail="unknown tagger")
    if req.on_conflict not in ("overwrite", "skip"):
        raise HTTPException(status_code=400, detail="on_conflict must be overwrite or skip")
    if not Path(req.path).is_dir():
        raise HTTPException(status_code=400, detail="path not found or not a directory")
    job_id, total = get_pathtag_queue().submit(
        req.path, req.model, req.gen_th, req.char_th,
        req.use_char, req.recursive, req.on_conflict)
    return {"job_id": job_id, "total": total}


@router.get("/{job_id}/events")
async def pathtag_events(job_id: str):
    state = get_pathtag_queue().state(job_id)
    if state is None:
        return StreamingResponse(
            iter(["data: " + json.dumps(
                {"type": "done", "done": 0, "total": 0, "ok": 0, "skipped": 0, "errors": 0}
            ) + "\n\n"]),
            media_type="text/event-stream",
        )

    async def gen():
        idx = 0
        while True:
            while idx < len(state.events):
                yield "data: " + json.dumps(state.events[idx]) + "\n\n"
                if state.events[idx].get("type") == "done":
                    return
                idx += 1
            if state.finished and idx >= len(state.events):
                return
            ev = asyncio.Event()
            state.waiters.append(ev)
            await ev.wait()

    return StreamingResponse(gen(), media_type="text/event-stream")
```

```typescript
// 路径打标：提交文件夹路径，后端异步反推并写同名 .txt
export async function startPathTag(payload: {
  path: string; model: string; gen_th: number; char_th: number;
  use_char: boolean; recursive: boolean; on_conflict: 'overwrite' | 'skip';
}): Promise<{ job_id: string; total: number }> {
  return fetch(`${base}/api/pathtag/start`, {
    method: 'POST', headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
  }).then(r => r.json())
}

// 订阅路径打标进度（SSE），返回 EventSource 句柄
export function subscribePathTag(
  jobId: string,
  onEvent: (e: any) => void,
  onDisconnect?: () => void,
) {
  const es = new EventSource(`${base}/api/pathtag/${jobId}/events`)
  es.onmessage = (m) => {
    const data = JSON.parse(m.data)
    onEvent(data)
    if (data.type === 'done') es.close()
  }
  es.onerror = () => { es.close(); onDisconnect?.() }
  return es
}
```

```typescript
import { reactive } from 'vue'
import { startPathTag, subscribePathTag } from '../api/client'

export interface PathTagLogItem { name: string; status: 'ok' | 'skip' | 'error'; msg?: string }

interface PathTagState {
  phase: 'idle' | 'running' | 'done' | 'error'
  total: number
  done: number
  ok: number
  skipped: number
  failed: number
  current: string
  jobId: string | null
  log: PathTagLogItem[]
}

const state = reactive<PathTagState>({
  phase: 'idle', total: 0, done: 0, ok: 0, skipped: 0, failed: 0,
  current: '', jobId: null, log: [],
})

// EventSource 由模块持有，不绑组件生命周期（同 useBatch）
let es: { close: () => void } | null = null

function isBusy() {
  return state.phase === 'running'
}

function reset() {
  state.phase = 'idle'
  state.total = state.done = state.ok = state.skipped = state.failed = 0
  state.current = ''
  state.jobId = null
  state.log = []
  es?.close()
  es = null
}

async function start(payload: {
  path: string; model: string; gen_th: number; char_th: number;
  use_char: boolean; recursive: boolean; on_conflict: 'overwrite' | 'skip';
}) {
  reset()
  state.phase = 'running'
  const r = await startPathTag(payload)
  state.total = r.total
  state.jobId = r.job_id
  // 同步订阅：消除 start 返回后、subscribe 赋值前的竞态（SSE 回调异步触发，本同步赋值不冲突）
  es = subscribePathTag(
    r.job_id,
    (ev) => {
      if (ev.type === 'progress') {
        state.done = ev.done
        state.current = ev.current || ''
        if (ev.status === 'skip') {
          state.skipped++
          state.log.push({ name: ev.current, status: 'skip' })
        } else {
          state.ok++
          state.log.push({ name: ev.current, status: 'ok' })
        }
      } else if (ev.type === 'error') {
        state.failed++
        state.log.push({ name: ev.current, status: 'error', msg: ev.message })
      } else if (ev.type === 'done') {
        state.done = ev.done
        state.phase = 'done'
        es?.close()
        es = null
      }
    },
    () => {
      // 断连：避免 phase 永久卡 running。守卫 running 避免覆盖 done 正常结束。
      if (state.phase === 'running') {
        state.phase = 'error'
        es?.close()
        es = null
      }
    },
  )
}

export function usePathTag() {
  return { state, isBusy, start, reset }
}
```
